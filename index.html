Basic with RWD


var basicSwipers = new PotatoSwiper( document.querySelectorAll( '.slider__items--basic' ), {
  loop: true,
  selector: {
    prev: '.slider__nav--prev',
    next: '.slider__nav--next',
    dot: '.slider__dot'
  },
  class: {
    activeDot: 'slider__dot--active'
  },
  gap: 20,
  items: 1,
  rwd: {
    420: {
      items: 2
    },
    640: {
      items: 3
    },
    960: {
      items: 4
    }
  }
} )

Multiple perItem with large and RWD


var multiPerItemSwipers = new PotatoSwiper( document.querySelectorAll( '.slider__items--perItem' ), {
  loop: true,
  selector: {
    prev: '.slider__nav--prev',
    next: '.slider__nav--next',
    dot: '.slider__dot',
    large: '.slider__item--large'
  },
  class: {
    activeDot: 'slider__dot--active'
  },
  items: 1,
  perItem: 1,
  rwd: {
    420: {
      items: 2
    },
    640: {
      items: 3,
      perItem: 2,
    },
    960: {
      items: 4,
      perItem: 2,
    }
  }
} )

Default settings


{
  nameSpace: 'potatoSwiper',
  autoInit: true,
  loop: true,
  items: 1,
  perItem: 1,
  autoWidth: false,
  scopeEl: rootEl.parentElement,
  prevSelector: '[data-ps-prev]',
  nextSelector: '[data-ps-next]',
  dotSelector: '[data-ps-dot]',
  class: {},
  gap: 0, // px
  largeSize: 2,
  largeSelector: '[data-ps-large]',
  duration: 500, // ms
  threshold: 20, // px
  easing: 'easeOutCubic',
  rwdMobileFirst: true,
  rwd: {}
}
